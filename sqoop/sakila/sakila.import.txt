## import all payments from sakila database to hdfs
sqoop-import --connect jdbc:mysql://iop-bi-master.imdemocloud.com:3306/sakila --username root \
--query "SELECT p.payment_id, p.rental_id, p.amount, p.payment_date, p.last_update, c.customer_id, concat(c.first_name, ' ', c.last_name) customer_name, s.staff_id, concat(s.first_name, ' ', s.last_name) staff_name, r.rental_date, r.return_date FROM payment p  join customer c on c.customer_id = p.customer_id  join staff s on s.staff_id = p.staff_id left join rental r on r.rental_id = p.rental_id where \$CONDITIONS" \
--target-dir /user/okmich20/output/handsontrain/sqoop/sakila/payment  \
--split-by payment_id --num-mappers 2 \
-P

## import all films (denormalized) from sakila db to hdfs compressing the output
sqoop-import --connect jdbc:mysql://iop-bi-master.imdemocloud.com:3306/sakila --username root \
--table v_film --target-dir /user/okmich20/output/handsontrain/sqoop/sakila/film  --compress \
-m 1 --split-by film_id -P
